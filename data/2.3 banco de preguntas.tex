\section{2.3) Otras Ecuaciones Diferenciales Ordinarias}

1)
\[
e^{y'} = x
\]
\textbf{Rpta: }
\[
y = x(\ln x - 1) + C
\]

\vspace{10pt}

2)
\[
\tan y' = 0
\]
\textbf{Rpta: }
\[
y = \pi n\,x + C
\]

\vspace{10pt}

3)
\[
e^{y'} = e^{4y'} + 1,\quad y \text{ acotada cuando } x \to +\infty
\]
\textbf{Rpta: }
\[
y = 0
\]

\vspace{10pt}

4)
\[
\sin y' = x
\]
\textbf{Rpta: }
\[
y = x\,\arcsin x - \sqrt{1 - x^{2}} + n\pi x
\]

\vspace{10pt}

5)
\[
x^{2}y' + \cos(2y) = 1,\quad y \to \tfrac{16\pi}{3}\ \text{cuando } x \to +\infty
\]
\textbf{Rpta: }
\[
y = \arctan\!\left(\frac{2}{x} + \frac{1}{\sqrt{3}}\right) + 3\pi
\]

\vspace{10pt}

6)
\[
(x+1)y' = y - 1,\quad y \text{ acotada cuando } x \to +\infty
\]
\textbf{Rpta: }
\[
y = 1
\]

\vspace{10pt}

7)
\[
y' = 2x(\pi + y),\quad y \text{ acotada cuando } x \to \infty
\]
\textbf{Rpta: }
\[
y = -\pi
\]

\vspace{10pt}

8)
\[
x^{3}y' - \sin y = 1,\quad y \to 5\pi,\ x \to \infty
\]
\textbf{Rpta: }
\[
y = 2\arctan\!\left(1 - \frac{1}{2x^{2}}\right) + \frac{9\pi}{2}
\]

\vspace{10pt}

9)
\[
y = \ln\!\left(\frac{dy}{dx}\right)
\]
\textbf{Rpta: }
\[
y = -\ln(C - x)
\]

\vspace{10pt}

10)
\[
(1+x^{2})y' - \tfrac{1}{2}\cos^{2}(2y) = 0,\quad y \to \tfrac{7\pi}{2},\ x \to -\infty
\]
\textbf{Rpta: }
\[
y = \tfrac{1}{2}\arctan\!\left(\tfrac{\pi}{2} + \arctan x\right) + \tfrac{7\pi}{2}
\]
