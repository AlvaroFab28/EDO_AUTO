\section*{II. Encontrar la solución particular de la ecuación diferencial, mediante las condiciones dadas.}

\]

\[

\vspace{10pt}

5)
\[

\vspace{10pt}

\section*{22) Hallar \(Y\) si:}

a)
\[
\int_{a}^{x} y\,dx = K\,(y^3 - b^3)
\]
\textbf{Rpta: }
\[
3k\,y^2 - 2x = 3k b^2 - 2a
\]

\vspace{10pt}

b)
\[
\int_{a}^{x} y\,dx = K\,(y - b)
\]
\textbf{Rpta: }
\[
y = e^{\frac{x - a}{k}}
\]

\vspace{10pt}

c)
\[
\int_{a}^{x} y\,dx = K\,(y^2 - b^2)
\]
\textbf{Rpta: }
\[
y = \frac{x - a \pm 2kb}{2k}
\]

\vspace{10pt}

d)
\[
\int_{a}^{x} y^2\,dx = K\,(y - b)
\]
\textbf{Rpta: }
\[
y \!\left( x - a - \frac{k}{b} \right) + k = 0
\]

\vspace{10pt}

e)
\[
\int_{a}^{x} y^2\,dx = K\,(y^2 - b^2)
\]
\textbf{Rpta: }
\[
x - a = 2k \ln \!\left( \frac{y}{b} \right)
\]

\vspace{10pt}

f)
\[
\int_{a}^{x} x^2\,dy = x^3 (y - b)
\]
\textbf{Rpta: }
\[
(2y - 3b) x^2 = -a^2 b
\]

\vspace{10pt}

g)
\[
\int_{a}^{x} x^6 y^2\,dx = x^7 (y^2 - b^2)
\]
\textbf{Rpta: }
\[
(6y^2 - 7b^2) x^6 + a^6 b^2 = 0
\]
